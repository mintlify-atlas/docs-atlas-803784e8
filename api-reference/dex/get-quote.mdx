---
title: "Get Quote"
description: "Get a price quote for a token swap without executing the transaction"
---

## Method

```typescript
async getQuote(params: QuoteParams): Promise<APIResponse<QuoteData>>
```

Retrieves pricing information and routing details for a potential token swap. This method does not execute a transaction.

## Parameters

<ParamField path="params" type="QuoteParams" required>
  Quote request parameters

  <Expandable title="properties">
    <ParamField path="chainIndex" type="string">
      Chain identifier (e.g., "1" for Ethereum, "501" for Solana)
    </ParamField>

    <ParamField path="fromTokenAddress" type="string" required>
      Contract address of the token to swap from
    </ParamField>

    <ParamField path="toTokenAddress" type="string" required>
      Contract address of the token to swap to
    </ParamField>

    <ParamField path="amount" type="string" required>
      Amount to swap (in token's smallest unit)
    </ParamField>

    <ParamField path="slippagePercent" type="string" required>
      Maximum acceptable slippage (0-1, e.g., "0.005" for 0.5%)
    </ParamField>

    <ParamField path="userWalletAddress" type="string">
      User's wallet address
    </ParamField>

    <ParamField path="dexIds" type="string">
      Comma-separated list of DEX IDs to route through
    </ParamField>

    <ParamField path="directRoute" type="boolean">
      Whether to use direct routing only
    </ParamField>

    <ParamField path="priceImpactProtectionPercent" type="string">
      Maximum acceptable price impact (0-1)
    </ParamField>

    <ParamField path="feePercent" type="string">
      Referral fee percentage
    </ParamField>
  </Expandable>
</ParamField>

## Example

```typescript
import { OKXClient } from '@okxweb3/okx-api';

const client = new OKXClient({
  apiKey: 'your-api-key',
  secretKey: 'your-secret-key',
  apiPassphrase: 'your-passphrase',
  projectId: 'your-project-id'
});

const quote = await client.dex.getQuote({
  chainIndex: '1',
  fromTokenAddress: '0xEeeeeEeeeEeEeeEeEeEeeEEEeeeeEeeeeeeeEEeE',
  toTokenAddress: '0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48',
  amount: '1000000000000000000', // 1 ETH
  slippagePercent: '0.005',
  userWalletAddress: '0x742d35Cc6634C0532925a3b844Bc454e4438f44e'
});

console.log('Quote:', quote.data[0]);
console.log('Expected output:', quote.data[0].toTokenAmount);
console.log('Price impact:', quote.data[0].priceImpactPercent);
```
