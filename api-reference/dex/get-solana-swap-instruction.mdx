---
title: getSolanaSwapInstruction
description: Get Solana swap instructions for manual transaction construction
---

Get raw Solana swap instructions that can be manually constructed and signed. This method provides more control over transaction construction compared to `executeSwap`.

## Method signature

```typescript
client.dex.getSolanaSwapInstruction(params: SwapParams): Promise<APIResponseSingle<SolanaSwapInstructionData>>
```

## Parameters

<ParamField path="chainIndex" type="string" required>
  Chain identifier. For Solana mainnet: `"501"`
</ParamField>

<ParamField path="fromTokenAddress" type="string" required>
  Source token address (SPL token or `So11111111111111111111111111111111111111112` for SOL)
</ParamField>

<ParamField path="toTokenAddress" type="string" required>
  Destination token address (SPL token or wrapped SOL)
</ParamField>

<ParamField path="amount" type="string" required>
  Amount to swap in smallest unit (lamports for SOL, base units for SPL tokens)
</ParamField>

<ParamField path="userWalletAddress" type="string" required>
  Solana wallet address executing the swap
</ParamField>

<ParamField path="slippagePercent" type="string">
  Maximum acceptable slippage as a decimal (e.g., `"0.005"` for 0.5%)
  
  Either `slippagePercent` or `autoSlippage` must be provided.
</ParamField>

<ParamField path="autoSlippage" type="boolean">
  Enable automatic slippage calculation based on market conditions
  
  When enabled, `maxAutoSlippagePercent` must also be provided.
</ParamField>

<ParamField path="maxAutoSlippagePercent" type="string">
  Maximum slippage when using auto slippage (e.g., `"0.05"` for 5%)
  
  Required when `autoSlippage` is `true`.
</ParamField>

<ParamField path="computeUnitPrice" type="string">
  Compute unit price in micro-lamports for priority fees
</ParamField>

<ParamField path="computeUnitLimit" type="string">
  Maximum compute units to use for the transaction
</ParamField>

## Response

Returns `APIResponseSingle<SolanaSwapInstructionData>` containing:

<ResponseField name="code" type="string">
  Response code (`"0"` for success)
</ResponseField>

<ResponseField name="msg" type="string">
  Response message
</ResponseField>

<ResponseField name="data" type="SolanaSwapInstructionData">
  Swap instruction data
  
  <Expandable title="SolanaSwapInstructionData properties">
    <ResponseField name="addressLookupTableAccount" type="string[]">
      Address lookup table accounts for transaction size optimization
    </ResponseField>
    
    <ResponseField name="instructionLists" type="SolanaInstructionItem[]">
      List of Solana instructions to execute
      
      <Expandable title="SolanaInstructionItem properties">
        <ResponseField name="data" type="string">
          Instruction data (base64 encoded)
        </ResponseField>
        
        <ResponseField name="accounts" type="SolanaInstructionAccount[]">
          Account metas for the instruction
        </ResponseField>
        
        <ResponseField name="programId" type="string">
          Program ID to invoke
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="routerResult" type="RouterResult">
      Quote information including expected amounts and price impact
    </ResponseField>
    
    <ResponseField name="tx" type="object">
      Transaction metadata
      
      <Expandable title="Transaction properties">
        <ResponseField name="from" type="string">
          Source wallet address
        </ResponseField>
        
        <ResponseField name="to" type="string">
          Destination token address
        </ResponseField>
        
        <ResponseField name="minReceiveAmount" type="string">
          Minimum tokens to receive after slippage
        </ResponseField>
        
        <ResponseField name="slippagePercent" type="string">
          Applied slippage percentage
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## Example

```typescript
import { OKXDexClient } from '@okx-dex/okx-dex-sdk';

const client = new OKXDexClient({
  apiKey: process.env.OKX_API_KEY!,
  secretKey: process.env.OKX_SECRET_KEY!,
  apiPassphrase: process.env.OKX_API_PASSPHRASE!,
  projectId: process.env.OKX_PROJECT_ID!
});

// Get Solana swap instructions
const instructionData = await client.dex.getSolanaSwapInstruction({
  chainIndex: '501',
  fromTokenAddress: 'So11111111111111111111111111111111111111112', // SOL
  toTokenAddress: 'EPjFWdd5AufqSSqeM2qN1xzybapC8G4wEGGkZwyTDt1v', // USDC
  amount: '1000000000', // 1 SOL
  slippagePercent: '0.005',
  userWalletAddress: walletAddress,
  computeUnitPrice: '1000'
});

console.log('Instructions:', instructionData.data.instructionLists);
console.log('Expected output:', instructionData.data.routerResult.toTokenAmount);
```

## Error handling

```typescript
try {
  const instructionData = await client.dex.getSolanaSwapInstruction({
    chainIndex: '501',
    fromTokenAddress,
    toTokenAddress,
    amount,
    slippagePercent: '0.005',
    userWalletAddress
  });
  
  // Manually construct and sign transaction
  // ...
  
} catch (error: any) {
  if (error.message.includes('slippage')) {
    console.error('Invalid slippage configuration');
  } else if (error.message.includes('Empty instruction data')) {
    console.error('No swap route available');
  } else {
    console.error('Failed to get instructions:', error.message);
  }
}
```
